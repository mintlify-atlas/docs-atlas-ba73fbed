---
title: "Get Products by Category"
api: "GET /products/category/:categoryId"
description: "Retrieve all products belonging to a specific category"
---

## Path Parameters

<ParamField path="categoryId" type="number" required>
  The unique identifier of the category
</ParamField>

## Response

Returns an array of products belonging to the specified category, ordered by creation date (newest first).

<ResponseField name="products" type="Product[]">
  <Expandable title="Product properties">
    <ResponseField name="id" type="number">
      Unique identifier for the product
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Product name
    </ResponseField>
    
    <ResponseField name="slug" type="string">
      URL-friendly unique identifier
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Product description
    </ResponseField>
    
    <ResponseField name="price" type="decimal">
      Product price (up to 10 digits, 2 decimal places)
    </ResponseField>
    
    <ResponseField name="isActive" type="boolean">
      Whether the product is currently active
    </ResponseField>
    
    <ResponseField name="categoryId" type="number">
      ID of the category this product belongs to
    </ResponseField>
    
    <ResponseField name="createdAt" type="datetime">
      Timestamp when the product was created
    </ResponseField>
    
    <ResponseField name="updatedAt" type="datetime">
      Timestamp when the product was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET https://api.kiototeteria.com/products/category/1
```

```javascript JavaScript
const response = await fetch('https://api.kiototeteria.com/products/category/1');
```

</RequestExample>
